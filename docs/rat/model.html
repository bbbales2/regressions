 56class CodeExecutor(NodeWalker):
 57    traced_values: Dict[ast.ModelBase, Any]
 58    parameters: Dict[str, Any]
 59    left_side_of_sampling: Union[None, ast.ModelBase] = field(default=None)
 60
 61    def walk_Statement(self, node: ast.Statement):
 62        if node.op == "~":
 63            self.left_side_of_sampling = node.left
 64            return_value = self.walk(node.right)
 65            self.left_side_of_sampling = None
 66            return return_value
 67        else:
 68            raise ExecuteException(f"{node.op} operator not supported in {type(self)}", node)
 69
 70    def walk_Binary(self, node: ast.Binary):
 71        left = self.walk(node.left)
 72        right = self.walk(node.right)
 73        return eval(f"left {node.op} right")
 74
 75    def walk_IfElse(self, node: ast.IfElse):
 76        predicate = self.traced_values[node.predicate]
 77        return jax.lax.cond(predicate, lambda: self.walk(node.left), lambda: self.walk(node.right))
 78
 79    def walk_FunctionCall(self, node: ast.FunctionCall):
 80        argument_list = []
 81
 82        if self.left_side_of_sampling:
 83            argument_list += [self.walk(self.left_side_of_sampling)]
 84
 85        if node.arglist:
 86            argument_list += [self.walk(arg) for arg in node.arglist]
 87
 88        return getattr(math, node.name)(*argument_list)
 89
 90    def walk_Variable(self, node: ast.Variable):
 91        if node in self.traced_values:
 92            if node.name in self.parameters:
 93                trace = self.traced_values[node]
 94                return self.parameters[node.name][trace]
 95            else:
 96                return self.traced_values[node]
 97        else:
 98            return self.parameters[node.name]

101        return node.value

115    def walk_Statement(self, node: ast.Statement):
116        if node.op != "=":
117            return
118
119        primary_node = compiler.get_primary_ast_variable(node)
120        primary_name = primary_node.name
121        primary_variable = self.variable_table[primary_name]
122
123        if node.left.name != primary_name:
124            msg = f"Not Implemented Error: The left hand side of assignment must be the primary variable for now"
125            raise AstException("computing transformed parameters", msg, node)
126
127        dependency_finder = CodeExecutorDependencyFinder(self.trace_table)
128        dependency_finder.walk(node)
129        traces_used = dependency_finder.traces_used
130
131        traced_values = tuple(self.trace_table[traced_node].array for traced_node in traces_used)
132
133        def mapper(*mapper_arguments):
134            executor = CodeExecutor(dict(zip(traces_used, mapper_arguments)), self.parameters)
135            return executor.walk(node.right)
136
137        # We can't be overwriting parameters
138        assert primary_name not in self.parameters
139
140        if primary_variable.argument_count > 0:
141            self.parameters[primary_name] = jax.vmap(mapper)(*traced_values)
142        else:
143            self.parameters[primary_name] = mapper(*traced_values)

216    def size(self):

219    @partial(jax.jit, static_argnums=(0,))
220    def constrain(self, unconstrained_parameter_vector: numpy.ndarray):
221        constraint_finder = ConstraintFinder(self.variable_table, self.trace_table)
222        constraint_finder.walk(self.program)
223        jacobian_adjustments = 0.0
224        parameters = {}
225        used = 0
226
227        for name in self.variable_table:
228            record = self.variable_table[name]
229
230            if not isinstance(record, SampledVariableRecord):
231                continue
232
233            # This assumes that unconstrained parameter indices for a parameter is allocated in a contiguous fashion.
234            if len(record.subscripts) > 0:
235                unconstrained = unconstrained_parameter_vector[used : used + len(record)]
236                used += len(record)
237            else:
238                unconstrained = unconstrained_parameter_vector[used]
239                used += 1
240
241            if name in constraint_finder.constraints:
242                variable_constraints = constraint_finder.constraints[name]
243
244                if "lower" in variable_constraints and "upper" not in variable_constraints:
245                    constrained, jacobian_adjustment = constraints.lower(unconstrained, variable_constraints["lower"])
246                elif "lower" not in variable_constraints and "upper" in variable_constraints:
247                    constrained, jacobian_adjustment = constraints.upper(unconstrained, variable_constraints["upper"])
248                else:  # "lower" in constraints and "upper" in constraints:
249                    constrained, jacobian_adjustment = constraints.finite(
250                        unconstrained, variable_constraints["lower"], variable_constraints["upper"]
251                    )
252
253                jacobian_adjustments += jax.numpy.sum(jacobian_adjustment)
254            else:
255                constrained = unconstrained
256
257            parameters[name] = constrained
258
259        return jacobian_adjustments, parameters
260
261    @partial(jax.jit, static_argnums=(0,))
262    def constrain_and_transform(self, unconstrained_parameter_vector: numpy.ndarray):
263        jacobian, parameters = self.constrain(unconstrained_parameter_vector)
264
265        transform = TransformedParametersFunctionGenerator(self.variable_table, self.trace_table, parameters)
266        transform.walk(self.program)
267
268        return jacobian, parameters

270    @partial(jax.jit, static_argnums=(0, 2))
271    def log_density(self, unconstrained_parameter_vector: numpy.ndarray, include_jacobian: bool = True):
272        jacobian, parameters = self.constrain(unconstrained_parameter_vector)
273
274        # Modify parameters in place
275        transform = TransformedParametersFunctionGenerator(self.variable_table, self.trace_table, parameters)
276        transform.walk(self.program)
277
278        likelihood = EvaluateDensityWalker(self.variable_table, self.trace_table, parameters)
279        target = likelihood.walk(self.program)
280
281        return target + (jacobian if include_jacobian else 0.0)

294    def __init__(self, model_string: str, data: Union[pandas.DataFrame, Dict[str, pandas.DataFrame]], max_trace_iterations: int = 50):
295        """
296        Create a model from some data (`data`) and a model (specified as a string, `model_string`).
297
298        If compile_path is not None, then write the compiled model to the given path (will only overwrite
299        existing files if the overwrite flag is true)
300
301        The parsed_lines argument is for creating a model from an intermediate representation -- likely
302        deprecated soon.
303
304        max_trace_iterations determines how many tracing iterations the program will do to resolve
305        parameter domains
306        """
307        data_dict = {}
308
309        match data:
310            case pandas.DataFrame():
311                data_dict["__default"] = data.reset_index().copy()
312            case dict():
313                for key, value in data.items():
314                    data_dict[key] = value.reset_index().copy()
315            case _:
316                raise Exception("Data must either be pandas data frames or a dict of pandas dataframes")
317
318        # Parse the model to get AST
319        semantics = ModelBuilderSemantics()
320        parser = RatParser(semantics=semantics)
321        # TODO: This lambda is just here to make sure pylance formatting works -- should work
322        # without it as well
323        self.program = (lambda: parser.parse(model_string))()
324
325        # Compile the model
326        rat_compiler = RatCompiler(data_dict, self.program, max_trace_iterations=max_trace_iterations)
327        self.variable_table = rat_compiler.variable_table
328        self.trace_table = rat_compiler.trace_table
329
330        self.base_df_dict = {}
331
332        for variable_name in self.variable_table:
333            record = self.variable_table[variable_name]
334            # The base_dfs are used for putting together the output
335            if isinstance(record, DynamicVariableRecord):
336                rows = list(record.itertuples())
337                self.base_df_dict[variable_name] = pandas.DataFrame.from_records(rows, columns=record.subscripts)
