 33    def __init__(self, arguments: Dict[ast.Variable, Any] = None, parameters: Dict[str, Any] = None):
 34        self.arguments = arguments
 35        self.parameters = parameters
 36        self.left_side_of_sampling = None
 37        super().__init__()

 39    def walk_Statement(self, node: ast.Statement):
 40        if node.op == "~":
 41            self.left_side_of_sampling = node.left
 42            return_value = self.walk(node.right)
 43            self.left_side_of_sampling = None
 44            return return_value
 45        else:
 46            raise ExecuteException(f"{node.op} operator not supported in BaseCodeGenerator", node)
 47
 48    def walk_Binary(self, node: ast.Binary):
 49        left = self.walk(node.left)
 50        right = self.walk(node.right)
 51        return eval(f"left {node.op} right")
 52
 53    def walk_IfElse(self, node: ast.IfElse):
 54        predicate = self.arguments[node.predicate]
 55        return jax.lax.cond(predicate, lambda: self.walk(node.left), lambda: self.walk(node.right))
 56
 57    def walk_FunctionCall(self, node: ast.FunctionCall):
 58        argument_list = []
 59
 60        if self.left_side_of_sampling:
 61            argument_list += [self.walk(self.left_side_of_sampling)]
 62
 63        if node.arglist:
 64            argument_list += [self.walk(arg) for arg in node.arglist]
 65
 66        return getattr(math, node.name)(*argument_list)
 67
 68    def walk_Variable(self, node: ast.Variable):
 69        if node in self.arguments:
 70            if node.name in self.parameters:
 71                trace = self.arguments[node]
 72                return self.parameters[node.name][trace]
 73            else:
 74                return self.arguments[node]
 75        else:
 76            return self.parameters[node.name]

 78    def walk_Literal(self, node: ast.Literal):
 79        return node.value

 89    def walk_Program(self, node: ast.Program):
 90        # TODO -- there is some order required here!
 91        for statement in node.statements:
 92            self.walk(statement)
 93
 94    def walk_Statement(self, node: ast.Statement):
 95        if node.op != "=":
 96            return
 97
 98        primary_node = compiler.get_primary_ast_variable(node)
 99        primary_name = primary_node.name
100        primary_variable = self.variable_table[primary_name]
101
102        if node.left.name != primary_name:
103            msg = f"Not Implemented Error: The left hand side of assignment must be the primary variable for now"
104            raise AstException("computing transformed parameters", msg, node)
105
106        self.traced_nodes = []
107
108        self.walk(node.left)
109        self.walk(node.right)
110
111        arguments = tuple(self.subscript_table[traced_node].array for traced_node in self.traced_nodes)
112
113        def mapper(*mapper_arguments):
114            executor = CodeExecutor(dict(zip(self.traced_nodes, mapper_arguments)), self.parameters)
115            return executor.walk(node.right)
116
117        # We can't be overwriting parameters
118        assert primary_name not in self.parameters
119
120        if primary_variable.argument_count > 0:
121            self.parameters[primary_name] = jax.vmap(mapper)(*arguments)
122        else:
123            self.parameters[primary_name] = mapper(*arguments)
124
125    def walk_Variable(self, node: ast.Variable):
126        node_variable = self.variable_table[node.name]
127        if node_variable.argument_count > 0:
128            if node in self.subscript_table:
129                self.traced_nodes.append(node)

186    @partial(jax.jit, static_argnums=(0,))
187    def constrain(self, unconstrained_parameter_vector: numpy.ndarray):
188        parameters = {}
189        jacobian_adjustments = 0.0
190
191        for name in self.variable_table:
192            record = self.variable_table[name]
193
194            if record.variable_type != VariableType.PARAM:
195                continue
196
197            # This assumes that unconstrained parameter indices for a parameter is allocated in a contiguous fashion.
198            if len(record.subscripts) > 0:
199                unconstrained = unconstrained_parameter_vector[
200                                record.unconstrained_vector_start_index: record.unconstrained_vector_end_index + 1]
201            else:
202                unconstrained = unconstrained_parameter_vector[record.unconstrained_vector_start_index]
203
204            if record.constraint_lower > float("-inf") or record.constraint_upper < float("inf"):
205                if record.constraint_lower > float("-inf") and record.constraint_upper == float("inf"):
206                    constrained, jacobian_adjustment = constraints.lower(unconstrained, record.constraint_lower)
207                elif record.constraint_lower == float("inf") and record.constraint_upper < float("inf"):
208                    constrained, jacobian_adjustment = constraints.upper(unconstrained, record.constraint_upper)
209                else:  # if record.constraint_lower > float("-inf") and record.constraint_upper < float("inf"):
210                    constrained, jacobian_adjustment = constraints.finite(unconstrained, record.constraint_lower,
211                                                                          record.constraint_upper)
212
213                jacobian_adjustments += jax.numpy.sum(jacobian_adjustment)
214            else:
215                constrained = unconstrained
216
217            parameters[name] = constrained
218
219        return jacobian_adjustments, parameters
220
221    @partial(jax.jit, static_argnums=(0,))
222    def constrain_and_transform(self, unconstrained_parameter_vector: numpy.ndarray):
223        jacobian, parameters = self.constrain(unconstrained_parameter_vector)
224
225        transform = TransformedParametersFunctionGenerator(self.variable_table, self.subscript_table, parameters)
226        transform.walk(self.program)
227
228        return jacobian, parameters

230    @partial(jax.jit, static_argnums=(0, 2))
231    def log_density(self, unconstrained_parameter_vector: numpy.ndarray, include_jacobian: bool = True):
232        jacobian, parameters = self.constrain(unconstrained_parameter_vector)
233
234        # Modify parameters in place
235        transform = TransformedParametersFunctionGenerator(self.variable_table, self.subscript_table, parameters)
236        transform.walk(self.program)
237
238        likelihood = EvaluateDensityWalker(self.variable_table, self.subscript_table, parameters)
239        target = likelihood.walk(self.program)
240
241        return target + (jacobian if include_jacobian else 0.0)

243    def log_density_no_jac(self, unconstrained_parameter_vector: numpy.ndarray):
244        return self.log_density(unconstrained_parameter_vector, False)

246    def _prepare_draws_and_dfs(self, device_unconstrained_draws):
247        constrain_and_transform_jax = jax.jit(jax.vmap(jax.vmap(lambda x: self.constrain_and_transform(x))))
248
249        _, constrained_draws = constrain_and_transform_jax(device_unconstrained_draws)

251        # # Copy back to numpy arrays
252        return {name: numpy.array(draws) for name, draws in constrained_draws.items()}, self.base_df_dict

254    def __init__(self, model_string: str, data: Union[pandas.DataFrame, Dict[str, pandas.DataFrame]],
255                 max_trace_iterations: int = 50):
256        """
257        Create a model from some data (`data`) and a model (specified as a string, `model_string`).
258
259        If compile_path is not None, then write the compiled model to the given path (will only overwrite
260        existing files if the overwrite flag is true)
261
262        The parsed_lines argument is for creating a model from an intermediate representation -- likely
263        deprecated soon.
264
265        max_trace_iterations determines how many tracing iterations the program will do to resolve
266        parameter domains
267        """
268
269        # Parse the model to get AST
270        semantics = ModelBuilderSemantics()
271        parser = RatParser(semantics=semantics)
272        # TODO: This lambda is just here to make sure pylance formatting works -- should work
273        # without it as well
274        self.program = (lambda: parser.parse(model_string))()
275
276        # Compile the model
277        rat_compiler = RatCompiler(data, self.program, model_string, max_trace_iterations=max_trace_iterations)
278        self.variable_table, self.subscript_table = rat_compiler.compile()
279
280        self.base_df_dict = {}
281
282        for variable_name in self.variable_table:
283            record = self.variable_table[variable_name]
284            # The base_dfs are used for putting together the output
285            if record.variable_type != VariableType.DATA:
286                rows = list(record.itertuples())
287                self.base_df_dict[variable_name] = pandas.DataFrame.from_records(rows, columns=record.subscripts)
288
289    @staticmethod
290    def from_file(filename: str, data: Union[pandas.DataFrame, Dict[str, pandas.DataFrame]],
291                  max_trace_iterations: int = 50, ):
292        with open(filename) as f:
293            return Model(f.read(), data, max_trace_iterations)

261        existing files if the overwrite flag is true)
262
263        The parsed_lines argument is for creating a model from an intermediate representation -- likely

265
266        max_trace_iterations determines how many tracing iterations the program will do to resolve

268        """

275        self.program = (lambda: parser.parse(model_string))()
276
277        # Compile the model
278        rat_compiler = RatCompiler(data, self.program, model_string, max_trace_iterations=max_trace_iterations)
279        self.variable_table, self.subscript_table = rat_compiler.compile()
280
281        self.base_df_dict = {}
282
283        for variable_name in self.variable_table:
284            record = self.variable_table[variable_name]
285            # The base_dfs are used for putting together the output
286            if record.variable_type != VariableType.DATA:
287                rows = list(record.itertuples())
288                self.base_df_dict[variable_name] = pandas.DataFrame.from_records(rows, columns=record.subscripts)
